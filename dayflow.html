How to set up Apps Script (5 min)
  1. Open your Google Sheet → Extensions → Apps Script
  2. Delete existing code, paste the script shown below
  3. Click Deploy → New deployment → Web app
  4. Set "Who has access" to Anyone
  5. Copy the Web App URL and paste it above
function doPost(e) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  const data  = JSON.parse(e.postData.contents);
  if (sheet.getLastRow() === 0)
    sheet.appendRow(['Name','Start','End','Duration','Date','Placed At']);
  data.rows.forEach(r =>
    sheet.appendRow([r.name,r.start,r.end,r.duration,r.date,r.placedAt])
  );
  return ContentService.createTextOutput('OK');
}
